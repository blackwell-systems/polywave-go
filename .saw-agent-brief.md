// ValidateP1FileDisjointness checks that IMPLs in the same tier do not have
// overlapping file_ownership entries (P1 invariant). Returns ValidationError
// with Code "P1_FILE_OVERLAP" if the same file appears in multiple IMPLs.
func ValidateP1FileDisjointness(tier int, impls []*IMPLManifest) []ValidationError {
    var errs []ValidationError
    fileToIMPL := make(map[string]string) // file path -> IMPL slug

    for _, impl := range impls {
        for _, fo := range impl.FileOwnership {
            if existingIMPL, exists := fileToIMPL[fo.File]; exists {
                errs = append(errs, ValidationError{
                    Code:    "P1_FILE_OVERLAP",
                    Message: fmt.Sprintf("File %s owned by both %s and %s in tier %d (violates P1 disjoint ownership)",
                        fo.File, existingIMPL, impl.FeatureSlug, tier),
                    Section: "file_ownership",
                })
            } else {
                fileToIMPL[fo.File] = impl.FeatureSlug
            }
        }
    }
    return errs
}
```

### InterviewModeSpec

Formal protocol specification for interview mode (deterministic requirements gathering)

```
## E39: Interview Mode (Deterministic Requirements Gathering)

**Trigger:** User invokes `/saw interview "<description>"` or `sawtools interview "<description>"`

**Rule:** Orchestrator enters INTERVIEWING state and conducts structured Q&A:
1. Phases: overview, scope, requirements, interfaces, stories, review (6 phases)
2. State persistence: After each answer, write INTERVIEW-<slug>.yaml to docs/
3. Resume capability: `--resume docs/INTERVIEW-<slug>.yaml` continues from last question
4. Output: On completion, generate REQUIREMENTS.md suitable for /saw bootstrap or /saw scout

**State machine:**
- SCOUTING → INTERVIEWING (on `/saw interview` command)
- INTERVIEWING → SCOUT_PENDING (on interview completion, REQUIREMENTS.md written)

**INTERVIEW doc schema:**
```yaml
slug: <feature-slug>
description: <user input>
status: in_progress | complete
current_phase: overview | scope | requirements | interfaces | stories | review
phases:
  - phase: overview
    questions: [{id, text, answer, timestamp}]
  - phase: scope
    questions: [{id, text, answer, timestamp}]
requirements_path: docs/REQUIREMENTS.md (populated on completion)
```

**Related Rules:** Scout Agent (Scout.md), E16 (IMPL Doc Validation)

```
